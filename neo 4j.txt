
#1.Find cars from same country (pattern discovery)
      (c2:Car)-[:MADE_BY]->(b2:Brand)-[:BASED_IN]->(co)
WHERE c1 MATCH (c1:Car)-[:MADE_BY]->(b1:Brand)-[:BASED_IN]->(co:Country),
<> c2
RETURN co.name, c1.name, c2.name;





#2.Indirect reasoning (important)
MATCH (c:Car)-[:MADE_BY]->(:Brand)-[:BASED_IN]->(co:Country)
RETURN c.name, co.name;




#3.Find cars made by brands in USA
MATCH (c:Car)-[:MADE_BY]->(b:Brand)-[:BASED_IN]->(co:Country)
WHERE co.name = "USA"
RETURN c.name, b.name;
